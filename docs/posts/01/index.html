  • 아래 명령어를 입력하면서 뜻 생각해보기, 명령어 안에서 줄바꿈은 shift+엔터를 쳐야 함
description <- function(x){
    mx=mean(x)
    se=sd(x)/sqrt(length(x))
    med=median(x)
    mode=as.numeric(names(which.max(table(x))))
    sx=sd(x)
    vx=var(x)
    kurt=mean((x-mean(x))^4)/(mean((x-mean(x))^2))^2-3
    skew=mean((x-mean(x))^3)/(mean((x-mean(x))^2))^1.5
    range=max(x)-min(x)
    quantiles=quantile(x)
    iqr=IQR(x)
    cat("column1","\n",
    "평균",mx,"\n",
    "표준오차",se,"\n",
    "중앙값",med,"\n", 
    "최빈값",mode,"\n",
    "표준편차",sx,"\n",
    "분산",vx,"\n",
    "첨도",kurt,"\n",
    "왜도",skew,"\n",
    "범위",range,"\n",
    "사분위수",quantiles,"\n",
    "사분위수범위",iqr,"\n",
    "최솟값",min(x),"\n",
    "최댓값",max(x),"\n",
    "합",sum(x),"\n",
    "관측수",length(x),"\n")
}

광역자치단체별로 평균 인구수를 구하기

ssg_tbl <- sgg_raw |>
  janitor::clean_names(ascii = FALSE) %>%  
  mutate(인구_명 = parse_number(인구_명)) |>
  group_by(광역자치단체) |>
  summarise(평균인구수 = mean(인구_명, na.rm = TRUE)) |>
  arrange(desc(평균인구수))

범주형 데이터를 예쁜 그래프로 그리기
